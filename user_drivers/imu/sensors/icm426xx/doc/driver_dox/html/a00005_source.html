63 /*
64  * For parts with maximum ODR of 32 KHz,
65  * PLL is running at 19.2 MHz instead of the nominal 20.48 MHz
66  * Time resolution has to be scaled by 20.48/19.2=1.06666667
67  * To prevent floating point usage, one can use 32/30
68  *
69  * The decimator also need to know the maximum ODR.
70  */
71 #define PLL_SCALE_FACTOR_Q24 ((32UL << 24) / 30)
72 #define ICM_PART_DEFAULT_OIS_MODE ICM426XX_SENSOR_CONFIG2_OIS_MODE_32KHZ
73 #else
74 #define PLL_SCALE_FACTOR_Q24 (1UL << 24)
75 #define ICM_PART_DEFAULT_OIS_MODE ICM426XX_SENSOR_CONFIG2_OIS_MODE_8KHZ
